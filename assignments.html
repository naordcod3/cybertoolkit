Set up a Domain Controller (`Gil.com`) and configure Active Directory.

# Install Active Directory Services
Install-WindowsFeature -Name AD-Domain-Services -IncludeManagementTools

# Create a New Forest and Set Up the Domain
Install-ADDSForest -DomainName "Gil.com" -DomainNetBiosName "GIL" -InstallDns:$true -Force

# Create OUs
New-ADOrganizationalUnit -Name "Managers"
New-ADOrganizationalUnit -Name "HR"
New-ADOrganizationalUnit -Name "Finance"
New-ADOrganizationalUnit -Name "Marketing"
New-ADOrganizationalUnit -Name "Development"
New-ADOrganizationalUnit -Name "IT"

# Create Users in Each Department (Example for HR)
for ($i=1; $i -le 5; $i++) {
    New-ADUser -Name "HRUser$i" -AccountPassword (ConvertTo-SecureString "Pass123!" -AsPlainText -Force) -Enabled $true -Path "OU=HR,DC=Gil,DC=com"
}

# Create Security Groups and Add Users (Example for HR)
New-ADGroup -Name "HRGroup" -GroupCategory Security -GroupScope Global -Path "OU=HR,DC=Gil,DC=com"
Get-ADUser -Filter {Name -like "HRUser*"} | ForEach-Object {Add-ADGroupMember -Identity "HRGroup" -Members $_.SamAccountName}

# Join Windows 10 Clients to the Domain (Run on the Client Machines)
Add-Computer -DomainName "Gil.com" -Credential Gil\Admin -Restart

  • Block PowerShell for Marketing
    Create a GPO to disable PowerShell script execution for all users in the Marketing OU 
                            # Create the GPO
                        New-GPO -Name "Block_PowerShell_Marketing"
                        
                        # Link the GPO to the Marketing OU
                        New-GPLink -Name "Block_PowerShell_Marketing" -Target "OU=Marketing,DC=Gil,DC=com"
                        
                        # Disable PowerShell script execution via Registry in the GPO
                        Set-GPRegistryValue -Name "Block_PowerShell_Marketing" -Key "HKLM\Software\Policies\Microsoft\Windows\System" -ValueName "EnableScripts" -Type DWord -Value 0
                        
                        # Apply the GPO immediately
                        gpupdate /force

Set up a Windows Server environment with Active Directory, user groups, and storage management. 

# Step 1: Create Virtual Machines
# - 3 Windows 10 VMs
# - 2 Windows Server 2022 VMs (No GUI)
# - Install VMware Tools and take snapshots

# Step 2: Configure Networking (172.16.0.0/16)
# - Assign IP addresses based on topology
# - Ensure all machines can communicate

# Step 3: Promote the Domain Controller (172.16.0.100)
Install-WindowsFeature -Name AD-Domain-Services -IncludeManagementTools
Install-ADDSForest -DomainName "gil.com" -DomainNetBiosName "GIL" -InstallDns:$true -Force

# Step 4: Join All Machines to the Domain
Add-Computer -DomainName "gil.com" -Credential gil\Admin -Restart

# Step 5: Create Organizational Units and Users in AD
New-ADOrganizationalUnit -Name "Managers"
New-ADOrganizationalUnit -Name "Finance"
New-ADOrganizationalUnit -Name "HR"
New-ADOrganizationalUnit -Name "Marketing"
New-ADOrganizationalUnit -Name "IT"
New-ADOrganizationalUnit -Name "Development"

for ($i=1; $i -le 5; $i++) {
    New-ADUser -Name "HRUser$i" -AccountPassword (ConvertTo-SecureString "Pass123!" -AsPlainText -Force) -Enabled $true -Path "OU=HR,DC=gil,DC=com"
}

# Step 6: Configure Storage Server (172.16.0.110)
# - Add 6 HDDs (5GB each)
# - Create 6 partitions (4x 2.5GB, 2x 10GB)
New-VHD -Path "C:\Storage\Disk1.vhdx" -SizeBytes 5GB -Dynamic
New-Partition -DiskNumber 1 -UseMaximumSize -AssignDriveLetter | Format-Volume -FileSystem NTFS -NewFileSystemLabel "Storage1"

# Step 7: Set Up Security Groups & Permissions in AD
New-ADGroup -Name "HRGroup" -GroupCategory Security -GroupScope Global -Path "OU=HR,DC=gil,DC=com"
Get-ADUser -Filter {Name -like "HRUser*"} | ForEach-Object {Add-ADGroupMember -Identity "HRGroup" -Members $_.SamAccountName}

# Step 8: Share Storage Based on Department
New-SmbShare -Name "HR_Share" -Path "C:\HR" -FullAccess "HRGroup"

# Step 9: Verify Access
# - Log in with a user from each department
# - Ensure users can only access their assigned shared folder
  • Create 5 virtual machines (3 Windows 10, 2 Windows Server 2022 without GUI).
  • Install VMware Tools and take a snapshot for each machine.
  • Set up networking so all VMs can communicate within `172.16.0.0/16`.
  • Promote the `172.16.0.100` server as a Domain Controller for `gil.com`.
  • Join all machines, including servers, to the domain.
  • Rename computers as specified in the network diagram.
  • Create Organizational Units (OUs) in Active Directory.
  • Create 6 departments and add 5 users per department.
  • Set up a Storage Server (`172.16.0.110`) with six new HDDs.
  • Partition disks into 4x `2.5GB` and 2x `10GB` partitions.
  • Assign and configure shared storage with AD security groups.
  • Verify that each department's users can access only their allocated storage.
